```jsx
// src/components/Image.jsx
import { IKImage } from 'imagekitio-react'

export default function Image({ src, className, w, h, alt }) {
  return (
    <IKImage
      urlEndpoint={import.meta.env.VITE_IK_URL_ENDPOINT}
      path={src}
      className={className}
      loading="lazy"
      lqip={{ active: true, quality: 20 }}
      width={w}

      alt={alt}
    />
  )
```

### 5.2 Express App

```js
// backend/index.js
import express from 'express'
import cors from 'cors'
import connectDB from './lib/connectDB.js'
import userRouter from './routes/user.route.js'
import postRouter from './routes/post.route.js'
import commentRouter from './routes/comment.route.js'
import webhookRouter from './routes/webhook.route.js'

const app = express()

// CORS — must be before routes
app.use(cors({ origin: process.env.CLIENT_URL }))

// JSON body parser for all routes except webhook
app.use('/webhooks', webhookRouter)   // raw body needed for Svix verification
app.use(express.json())
app.use('/users', userRouter)
app.use('/posts', postRouter)
app.use('/comments', commentRouter)

// Global error handler (Express 5 — no try/catch needed in async routes)
app.use((err, req, res, next) => {
  res.status(err.status || 500).json({
    message: err.message || 'Something went wrong',
    stack: process.env.NODE_ENV === 'production' ? undefined : err.stack,
  })
})

app.listen(3000, async () => {
  await connectDB()
  console.log('Server running on port 3000')
})
```

<Callout type="warning" emoji="🚨">
    **Express 5 key change:** async route handlers automatically forward thrown errors to your error-handling middleware. You no longer need `try/catch` blocks or explicit `next(err)` calls in route handlers.
</Callout>

## Part 7 — Clerk Webhook (Sync Users to DB)

<Steps>

    ### Install Svix for signature verification

    ```bash
    npm install svix body-parser
    ```

    ### Expose your local server with ngrok

    ```bash
    ngrok http 3000
    # Copy the https URL → paste into Clerk Dashboard → Webhooks → Add Endpoint
    ```

    ### Select events

    Subscribe to `user.created`, `user.updated`, and `user.deleted` in the Clerk dashboard.

    ### Implement the webhook controller

    ```js
    // controllers/webhook.controller.js
    import { Webhook } from 'svix'
    import UserModel from '../models/user.model.js'

    export const clerkWebhook = async (req, res) => {
    const webhookSecret = process.env.CLERK_WEBHOOK_SECRET
    if (!webhookSecret) throw new Error('Webhook secret needed')

    const wh = new Webhook(webhookSecret)
    let event

    try {
    event = wh.verify(req.body, req.headers)
} catch {
    return res.status(400).json({ message: 'Webhook verification failed' })
}

    const { data, type } = event

    if (type === 'user.created') {
    const newUser = new UserModel({
    clerkUserId: data.id,
    username: data.username ?? data.email_addresses[0].email_address,
    email: data.email_addresses[0].email_address,
    img: data.profile_image_url,
})
    await newUser.save()
}

    if (type === 'user.deleted') {
    await UserModel.findOneAndDelete({ clerkUserId: data.id })
}

    res.status(200).json({ message: 'Webhook received' })
}
    ```

</Steps>

<Callout type="warning">
    Register the webhook route **before** `express.json()` so it receives the raw body. Svix needs the raw bytes to verify the HMAC signature.
</Callout>

## Part 9 — Data Fetching with TanStack React Query

### Install

```bash
npm install -f @tanstack/react-query axios
```

### Fetching a list with Infinite Scroll

```jsx
import { useInfiniteQuery } from '@tanstack/react-query'
import InfiniteScroll from 'react-infinite-scroll-component'
import axios from 'axios'

function PostList({ searchParams }) {
  const fetchPosts = async ({ pageParam = 1 }) => {
    const res = await axios.get(`${import.meta.env.VITE_API_URL}/posts`, {
      params: { page: pageParam, limit: 10, ...searchParams },
    })
    return res.data
  }

  const { data, fetchNextPage, hasNextPage, status } = useInfiniteQuery({
    queryKey: ['posts', searchParams],
    queryFn: fetchPosts,
    initialPageParam: 1,
    getNextPageParam: (lastPage) =>
      lastPage.hasMore ? lastPage.currentPage + 1 : undefined,
  })

  if (status === 'pending') return <p>Loading…</p>
  if (status === 'error')   return <p>Something went wrong.</p>

  const allPosts = data.pages.flatMap((page) => page.posts ?? [])

  return (
    <InfiniteScroll
      dataLength={allPosts.length}
      next={fetchNextPage}
      hasMore={!!hasNextPage}
      loader={<p>Loading more posts…</p>}
      endMessage={<p>All posts loaded.</p>}
    >
      {allPosts.map((post) => (
        <PostListItem key={post._id} post={post} />
      ))}
    </InfiniteScroll>
  )
}
```

### Mutations with Optimistic Updates

```jsx
import { useMutation, useQueryClient } from '@tanstack/react-query'
import { useAuth } from '@clerk/clerk-react'

function Comments({ postId }) {
  const { getToken } = useAuth()
  const queryClient = useQueryClient()

  const addComment = useMutation({
    mutationFn: async (newComment) => {
      const token = await getToken()
      return axios.post(
        `${import.meta.env.VITE_API_URL}/comments/${postId}`,
        newComment,
        { headers: { Authorization: `Bearer ${token}` } }
      )
    },
    onSuccess: () => queryClient.invalidateQueries({ queryKey: ['comments', postId] }),
    onError: (err) => toast.error(err.response.data),
  })

  // Show optimistic comment while pending
  return (
    <>
      {addComment.isPending && (
        <Comment comment={{ desc: addComment.variables.desc, createdAt: new Date() }} />
      )}
      {/* real comments list */}
    </>
  )
}
```

## Part 12 — Pagination on the Backend

```js
// controllers/post.controller.js
export const getPosts = async (req, res) => {
  const page  = parseInt(req.query.page)  || 1
  const limit = parseInt(req.query.limit) || 10

  const query = {}
  if (req.query.category) query.category   = req.query.category
  if (req.query.featured)  query.isFeatured = true
  if (req.query.search)    query.title      = { $regex: req.query.search, $options: 'i' }

  if (req.query.author) {
    const user = await UserModel.findOne({ username: req.query.author }).select('_id')
    if (!user) return res.status(404).json({ message: 'No posts found' })
    query.user = user._id
  }

  const sortObj = { createdAt: -1 }          // default: newest
  if (req.query.sort === 'oldest')   sortObj.createdAt = 1
  if (req.query.sort === 'popular')  { sortObj.visit = -1; delete sortObj.createdAt }
  if (req.query.sort === 'trending') {
    sortObj.visit = -1
    query.createdAt = { $gte: new Date(Date.now() - 7 * 24 * 60 * 60 * 1000) }
  }

  const [posts, totalPosts] = await Promise.all([
    PostModel.find(query)
      .populate('user', 'username img')
      .sort(sortObj)
      .skip((page - 1) * limit)
      .limit(limit),
    PostModel.countDocuments(query),
  ])

  res.status(200).json({
    posts,
    hasMore: page * limit < totalPosts,
    currentPage: page,
  })
}
```

## Part 14 — Search, Filters & URL State

```jsx
// PostListPage.jsx
import { useSearchParams } from 'react-router-dom'

const [searchParams, setSearchParams] = useSearchParams()

// Pass to fetch as plain object
const searchParamsObj = Object.fromEntries(searchParams.entries())

// Update a single param without losing others
const handleCategoryChange = (cat) => {
  setSearchParams({ ...searchParamsObj, category: cat })
}

const handleSortChange = (e) => {
  if (searchParams.get('sort') !== e.target.value)
    setSearchParams({ ...searchParamsObj, sort: e.target.value })
}
```

<Callout type="info">
    Passing `searchParams` as part of the `queryKey` ensures TanStack Query re-fetches automatically whenever the URL changes.
</Callout>

```js
export const createPost = async (req, res) => {
  const clerkUserId = req.auth?.userId
  if (!clerkUserId) return res.status(401).json({ message: 'Not authenticated' })

  const user = await UserModel.findOne({ clerkUserId })
  if (!user) return res.status(404).json({ message: 'User not found' })

  // Generate a unique slug from the title
  let slug = req.body.title.replace(/\s+/g, '-').toLowerCase()
  let existingPost = await PostModel.findOne({ slug })
  let counter = 2
  while (existingPost) {
    slug = `${slug}-${counter}`
    existingPost = await PostModel.findOne({ slug })
    counter++
  }

  const newPost = new PostModel({
    user: user._id,
    slug,
    ...req.body,
    img: req.body.img || '',
  })
  await newPost.save()
  res.status(201).json(newPost)
}
```

## Key Patterns Summary

<Callout type="info" emoji="💡">
    **Optimistic UI** — Before the server responds, immediately show the new comment (or toggled save state) using `mutation.isPending` + `mutation.variables`. TanStack Query's `invalidateQueries` then replaces the optimistic data with the real server response.
</Callout>

<Callout type="info" emoji="💡">
    **Visit counter middleware** — Use an Express middleware function (`increaseVisit`) on the `GET /posts/:slug` route that increments the `visit` field before the main handler runs. This keeps controllers clean.
</Callout>

<Callout type="info" emoji="💡">
    **Image sizing** — Always pass a `transformation` width equal to the maximum rendered size of the image (viewport width minus padding). This prevents serving 1 MB originals for 300 px thumbnails.
</Callout>

<Callout type="warning" emoji="⚠️">
    **React 19 RC compatibility** — Always add the `-f` (force) flag when installing npm packages to skip the peer-dependency check that would block installation against the RC version.
</Callout>
